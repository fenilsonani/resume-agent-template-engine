% Name and Contact Info
\newcommand{\personalinfo}[5]{
    \begin{center}
        {\Huge\bfseries\color{darkgray} #1}\\[0.5ex] % Name
        \normalsize #2 \textperiodcentered\ #3 \textperiodcentered\ \href{mailto:#4}{#4} \textperiodcentered\ \href{#5}{#5}
    \end{center}
    \vspace{1em}
}

}

% Experience Section Item
\newcommand{\experienceentry}[5]{ % Company, Title, Dates, Location, Details (list)
    \subsection*{#1 \hfill \small\color{lightgray} #3}
    \textit{#2 \hfill \small\color{lightgray} #4}\\[0.5ex]
    \begin{itemize*} % Compact itemize
        #5
    \end{itemize*}

}

% Education Section Item
\newcommand{\educationentry}[4]{ % Degree, Institution, Dates, Details
